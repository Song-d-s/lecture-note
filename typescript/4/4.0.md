## Abstract class
> cannot make an instance from it
```ts
abstract class User {
    constructor(
        private firstName: string,
        private lastName: string,
        // public nickname: string,
        // private nickname: string,
        protected nickname: string, // protected
    ) {}
    // abstract method - no implementation (no code)
    abstract getNickName(): void

    getFullName(){
        return `${this.firstName} ${this.lastName}`
    }
}

class Player extends User { // can extend abstract class
    getNickName() {
        console.log(this.nickname)
    }
}

// ðŸ’¡ Cannot create an instance directly from an abstract class!
// const coco = new User("coco", "las", "ì½”ì½”") ðŸ“Œ Error!
const coco = new Player("coco", "las", "ì½”ì½”")  // okay

coco.getFullName()
```
